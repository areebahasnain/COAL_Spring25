```
include irvine32.inc

.data
    a BYTE 00010001b  ; 11h
    b BYTE 00010000b  ; 10h
    _c BYTE 00110000b  ; 30h
    d BYTE 01000000b  ; 40h

.code
main proc
    mov al, a      
    add al, b      
    sub al, a      
    sub al, b      
    add al, _c      
    add al, d 

call DumpRegs  

exit
main endp
end main
```
