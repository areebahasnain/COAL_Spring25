```asm
INCLUDE Irvine32.inc

.data
Str1 BYTE "127&j~3#^&*#*#45^",0
index DWORD ?

.code
main PROC
    mov edx, OFFSET Str1
    call Scan_String
    mov index, eax

    ; print result
    mov eax, index
    call WriteDec

    exit
main ENDP

Scan_String PROC
    push ecx
    push edi

    cld
    mov edi, edx                ; string address
    mov al, '#'
    mov ecx, LENGTHOF Str1      
    repne scasb                 ; scan for '#'

    jz foundChar                ; ZF = 1 -> found
    mov eax, -1                 ; not found
    jmp done

foundChar:
    mov eax, LENGTHOF Str1
    sub eax, ecx                ; index = total - remaining

done:
    pop edi
    pop ecx
    ret
Scan_String ENDP


END main
```
